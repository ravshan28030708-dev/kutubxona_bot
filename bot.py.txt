class SimpleHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.end_headers()
        self.wfile.write(b"Bot ishlayapti...")

def run_server():
    port = int(os.environ.get("PORT", 8080))
    httpd = HTTPServer(('0.0.0.0', port), SimpleHandler)
    httpd.serve_forever()

def init_db():
    conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
    cursor = conn.cursor()
    cursor.execute('''CREATE TABLE IF NOT EXISTS books 
                      (id INTEGER PRIMARY KEY, title TEXT, author TEXT, genre TEXT, file_id TEXT)''')
    conn.commit()
    conn.close()

@bot.message_handler(commands=['start'])
def start(message):

    markup.add(*buttons)

                     "📚 **Xush kelibsiz!**\n\nPastdagi janrlardan birini tanlang:", 
                     reply_markup=markup, parse_mode="Markdown")

# --- JANR TANLANGANDA KITOB CHIQARISH ---
@bot.message_handler(func=lambda message: message.text in JANRLAR)
def show_books(message):
    conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
    cursor = conn.cursor()
    cursor.execute("SELECT title, author, file_id FROM books WHERE genre=?", (message.text,))
    books = cursor.fetchall()
    conn.close()

    if books:
        bot.send_message(message.chat.id, f"📂 **{message.text}** janridagi kitoblar:")
        for title, author, f_id in books:
            caption = f"📖 **Nomi:** {title}\n✍️ **Muallif:** {author}"
            bot.send_document(message.chat.id, f_id, caption=caption, parse_mode="Markdown")
    else:
        bot.send_message(message.chat.id, f"❌ Kechirasiz, **{message.text}** janrida kitob yo'q.")

@bot.message_handler(content_types=['document'])
def handle_pdf(message):
    chat_id = message.chat.id
    if chat_id in temp_data and chat_id == ADMIN_ID:
        data = temp_data[chat_id]
        f_id = message.document.file_id
        
        conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
        cursor = conn.cursor()
        cursor.execute("INSERT INTO books (title, author, genre, file_id) VALUES (?, ?, ?, ?)",
                       (data['title'], data['author'], data['genre'], f_id))
        conn.commit()
        conn.close()
        
        del temp_data[chat_id]
        bot.send_message(chat_id, "✅ Kitob muvaffaqiyatli saqlandi!")
