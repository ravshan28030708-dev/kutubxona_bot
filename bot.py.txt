def init_db():
    conn = sqlite3.connect('kutubxona.db')
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS books (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT,
            author TEXT,
            genre TEXT,
            file_id TEXT
        )
    ''')
    conn.commit()
    conn.close()

@bot.message_handler(func=lambda message: message.text == "📚 Kitoblar katalogi")
def show_genres(message):
    conn = sqlite3.connect('kutubxona.db')
    cursor = conn.cursor()
    cursor.execute("SELECT DISTINCT genre FROM books")
    genres = cursor.fetchall()
    conn.close()

    if not genres:
        bot.send_message(message.chat.id, "Kutubxonada hozircha kitoblar yo'q.")
        return

    markup = InlineKeyboardMarkup()
    for genre in genres:
        genre_name = genre[0]
        markup.add(InlineKeyboardButton(f"📁 {genre_name.capitalize()}", callback_data=f"genre_{genre_name}"))
    
    bot.send_message(message.chat.id, "Qaysi janrdagi kitoblarni o'qishni xohlaysiz?", reply_markup=markup)

@bot.callback_query_handler(func=lambda call: call.data.startswith("genre_"))
def show_books_by_genre(call):
    genre_name = call.data.split("_")[1]
    
    conn = sqlite3.connect('kutubxona.db')
    cursor = conn.cursor()
    cursor.execute("SELECT id, title, author FROM books WHERE genre=?", (genre_name,))
    books = cursor.fetchall()
    conn.close()

    markup = InlineKeyboardMarkup()
    for book in books:
        book_id, title, author = book
        markup.add(InlineKeyboardButton(f"📖 {title}", callback_data=f"book_{book_id}"))
    
    bot.edit_message_text(chat_id=call.message.chat.id, 
                          message_id=call.message.message_id, 
                          text=f"<b>{genre_name.capitalize()}</b> janridagi kitoblar:", 
                          parse_mode="HTML",
                          reply_markup=markup)

@bot.callback_query_handler(func=lambda call: call.data.startswith("book_"))
def send_book_file(call):
    book_id = call.data.split("_")[1]
    
    conn = sqlite3.connect('kutubxona.db')
    cursor = conn.cursor()
    cursor.execute("SELECT title, author, file_id FROM books WHERE id=?", (book_id,))
    book = cursor.fetchone()
    conn.close()

    if book:
        title, author, file_id = book
        bot.send_chat_action(call.message.chat.id, 'upload_document')
        bot.send_document(call.message.chat.id, file_id, caption=f"📚 Kitob: {title}\n✍️ Muallif: {author}\n\n🤖 @{bot.get_me().username} orqali yuklab olindi.")
    else:
        bot.answer_callback_query(call.id, "Fayl topilmadi!")

# Kitob qo'shish jarayonini saqlash uchun vaqtinchalik lug'at
temp_book_data = {}

@bot.message_handler(func=lambda message: message.text == "➕ Yangi kitob qo'shish (Admin)")
def add_book_start(message):
    if message.from_user.id != ADMIN_ID:
        bot.send_message(message.chat.id, "Sizda bu huquq yo'q!")
        return
    msg = bot.send_message(message.chat.id, "Kitob nomini kiriting:")
    bot.register_next_step_handler(msg, process_title_step)

def process_title_step(message):
    temp_book_data[message.from_user.id] = {'title': message.text}
    msg = bot.send_message(message.chat.id, "Muallif ismini kiriting:")
    bot.register_next_step_handler(msg, process_author_step)

def process_author_step(message):
    temp_book_data[message.from_user.id]['author'] = message.text
    msg = bot.send_message(message.chat.id, "Kitob janrini kiriting (masalan: Badiiy, Ilmiy, Dasturlash):")
    bot.register_next_step_handler(msg, process_genre_step)

def process_genre_step(message):
    temp_book_data[message.from_user.id]['genre'] = message.text.lower()
    msg = bot.send_message(message.chat.id, "Endi kitob faylini (PDF, EPUB yoki DOCX) yuboring:")
    bot.register_next_step_handler(msg, process_file_step)

def process_file_step(message):
    if message.document:
        file_id = message.document.file_id
        user_data = temp_book_data[message.from_user.id]
        
        # Bazaga saqlash
        conn = sqlite3.connect('kutubxona.db')
        cursor = conn.cursor()
        cursor.execute("INSERT INTO books (title, author, genre, file_id) VALUES (?, ?, ?, ?)",
                       (user_data['title'], user_data['author'], user_data['genre'], file_id))
        conn.commit()
        conn.close()
        
        bot.send_message(message.chat.id, f"✅ Kitob muvaffaqiyatli bazaga qo'shildi!\nNomi: {user_data['title']}")
        del temp_book_data[message.from_user.id] # Xotirani tozalash
    else:
        msg = bot.send_message(message.chat.id, "Iltimos, fayl yuboring! (PDF yoki DOCX formatda)")
        bot.register_next_step_handler(msg, process_file_step)
