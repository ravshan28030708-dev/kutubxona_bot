# --- 24/7 SERVER ---
class SimpleHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.end_headers()
        self.wfile.write(b"Bot is online")

def run_server():
    port = int(os.environ.get("PORT", 8080))
    httpd = HTTPServer(('0.0.0.0', port), SimpleHandler)
    httpd.serve_forever()

def init_db():
    conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
    cursor = conn.cursor()
    cursor.execute('''CREATE TABLE IF NOT EXISTS books 
                      (id INTEGER PRIMARY KEY, title TEXT, author TEXT, genre TEXT, file_id TEXT)''')
    conn.commit()
    conn.close()

# --- INLINE JANR BOSILGANDA ---
@bot.callback_query_handler(func=lambda call: call.data.startswith('janr_'))
def genre_selected(call):
    janr_nomi = call.data.split('_')[1]
    
    conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
    cursor = conn.cursor()
    cursor.execute("SELECT id, title FROM books WHERE genre=?", (janr_nomi,))
    books = cursor.fetchall()
    conn.close()

    markup = types.InlineKeyboardMarkup(row_width=1)
    if books:
        for b_id, title in books:
            markup.add(types.InlineKeyboardButton(f"📖 {title}", callback_data=f"kitob_{b_id}"))
        # Orqaga qaytish tugmasi
        markup.add(types.InlineKeyboardButton("🔙 Orqaga", callback_data="orqaga_katalog"))
        
        bot.edit_message_text(chat_id=call.message.chat.id, message_id=call.message.message_id,
                              text=f"**{janr_nomi}** janridagi kitoblar:", reply_markup=markup, parse_mode="Markdown")
    else:
        bot.answer_callback_query(call.id, f"{janr_nomi} janrida hozircha kitob yo'q.", show_alert=True)

@bot.message_handler(content_types=['document'])
def handle_pdf(message):
    chat_id = message.chat.id
    if chat_id in temp_data and chat_id == ADMIN_ID:
        data = temp_data[chat_id]
        f_id = message.document.file_id
        
        conn = sqlite3.connect('kutubxona.db', check_same_thread=False)
        cursor = conn.cursor()
        cursor.execute("INSERT INTO books (title, author, genre, file_id) VALUES (?, ?, ?, ?)",
                       (data['title'], data['author'], data['genre'], f_id))
        conn.commit()
        conn.close()
        
        del temp_data[chat_id]
        # Asosiy klaviaturani qaytaramiz
        reply_markup = types.ReplyKeyboardMarkup(resize_keyboard=True)
        reply_markup.add(types.KeyboardButton("📚 Kitoblar katalogi"))
        bot.send_message(chat_id, "✅ Kitob bazaga saqlandi!\n/start bosing yoki katalogni oching.", reply_markup=reply_markup)
